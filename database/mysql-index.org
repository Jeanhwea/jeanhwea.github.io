  构建测试表 ~author~, 建表语句如下
  #+BEGIN_SRC sql
    drop table if exists author;
    create table author (
      id int auto_increment primary key,
      name varchar(128) not null,
      age int,
      gender varchar(32),
      a int,
      b int,
      c int
    );
  #+END_SRC

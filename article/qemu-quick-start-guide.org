* 准备虚拟机
  #+BEGIN_SRC sh
    # 下载操作系统镜像
    wget https://mirrors.aliyun.com/centos/7.9.2009/isos/x86_64/CentOS-7-x86_64-Minimal-2009.iso

    # 创建硬盘镜像
    qemu-img create -f qcow2 hd001.qcow2 64G

    # 启动 qemu 模拟器
    qemu-system-x86_64 -hda hd001.qcow2 -cdrom CentOS-7-x86_64-Minimal-2009.iso -boot order=d -m 1024
  #+END_SRC

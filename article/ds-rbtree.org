* 红黑树定义
#+BEGIN_SRC dot :exports results :file ../static/image/2023/0222/122429.png
  digraph {
    graph [ordering="out"];
    node[shape=circle, style=filled, fontcolor=white, color=white];

    1  [fillcolor=red];
    2  [fillcolor=red];
    3  [fillcolor=black];
    4  [fillcolor=black];
    5  [fillcolor=black];
    12 [fillcolor=black];

    12 -> 1;
    12 -> 2;
    1 -> blank[style=invis];
    1 -> 3;
    3 -> 4;
    3 -> 5;
  }
#+END_SRC
